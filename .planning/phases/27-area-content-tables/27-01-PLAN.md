  - id SERIAL PRIMARY KEY

- CREATE INDEX idx_area_content_source_doc ON area_content(source_document_id)
- CREATE INDEX idx_area_content_metadata ON area_content USING gin(metadata) -- JSONB queries
- CREATE TRIGGER update_area_content_updated_at BEFORE UPDATE ON area_content FOR EACH ROW EXECUTE FUNCTION update_updated_at_column()
