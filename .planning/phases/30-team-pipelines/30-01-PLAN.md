```lisp
(in-package :af64.runtime.pipeline-definitions)

;;; ---------------------------------------------------------------------------
;;; Pipeline Definitions — load department and project pipeline definitions
;;; from area_content table, replacing hardcoded *pipeline-advancement*.
;;; Per D-01: pipeline definitions live in master_chronicle, not filesystem.
;;; Per D-14: accessor functions return same shapes as old defparameters.
;;; ---------------------------------------------------------------------------

(defvar *pipeline-cache* nil
  "Cached pipeline definitions loaded from area_content. List of parsed metadata hash-tables.
   Refreshed each tick via reload-pipeline-definitions.")

(defvar *advancement-cache* (make-hash-table :test #'equal)
  "Compound key cache: (pipeline-name . stage) -> (next-stage . next-assignee).
   Per D-07: (pipeline, stage) compound key eliminates stage name collisions.")

(defvar *prev-stage-cache* (make-hash-table :test #'equal)
  "Compound key cache: (pipeline-name . stage) -> predecessor-stage-name.
   Inverse of advancement: for each stage, what came before it.")

(defvar *stage-to-pipeline-cache* (make-hash-table :test #'equal)
  "Maps stage-name -> pipeline-name. For stages in multiple pipelines, stores last loaded.
   Primary lookup should use get-pipeline-type-for-task with goal_id chain.")

(defvar *fork-cache* (make-hash-table :test #'equal)
  "Maps (pipeline-name . stage) -> list of (fork-stage . fork-assignee).")

(defvar *participants-cache* (make-hash-table :test #'equal)
  "Maps pipeline-name -> list of unique assignee strings (for energy rewards).")

;;; ---------------------------------------------------------------------------
;;; Loading from area_content
;;; ---------------------------------------------------------------------------

(defun load-pipeline-definitions ()
  "Load all active pipeline definitions from area_content.
   Returns list of parsed metadata hash-tables."
  (handler-case
      (let* ((sql "SELECT metadata FROM area_content WHERE content_type = 'pipeline' AND status = 'active'")
             (rows (db-query sql)))
        (when rows
          (loop for i from 0 below (length rows)
                for row = (aref rows i)
                for meta = (gethash :METADATA row)
                when (hash-table-p meta)
                collect meta)))
    (error (e)
      (format t "[pipeline] Error loading definitions: ~a~%" e)
      nil)))

(defun build-advancement-from-pipeline (meta)
  "Build advancement entries from a single pipeline metadata hash-table.
   Populates *advancement-cache*, *prev-stage-cache*, *stage-to-pipeline-cache*,
   *fork-cache*, and *participants-cache*."
  (let* ((stages (gethash :STAGES meta))
         (pipeline-name (gethash :PIPELINE-NAME meta))
         (forks-raw (gethash :FORKS meta))
         (terminal (gethash :TERMINAL-ASSIGNEE meta))
         (stage-list (if (vectorp stages) (coerce stages 'list) stages))
         (forks-list (if (vectorp forks-raw) (coerce forks-raw 'list) forks-raw))
         (participants (make-hash-table :test #'equal)))
    (when (and pipeline-name stage-list)
      ;; Build sequential advancement from ordered stages
      ;; Stages are ordered by "order" field. For linear pipelines, iterate pairwise.
      ;; For forked pipelines, the stages array contains ALL stages in order,
      ;; and the forks array specifies where forks happen.
      ;; Build a map of from_stage -> primary_next from the forks first
      (let ((fork-primary-map (make-hash-table :test #'equal)))
        ;; For each fork, the from_stage's primary next is the first to_stage
        (dolist (fork forks-list)
          (let* ((from (gethash :FROM-STAGE fork))
                 (to-raw (gethash :TO-STAGES fork))
                 (to-list (if (vectorp to-raw) (coerce to-raw 'list) to-raw)))
            (when (and from to-list)
              ;; Primary next = first to_stage
              (let ((primary (first to-list)))
                (when (hash-table-p primary)
                  (setf (gethash from fork-primary-map)
                        (cons (gethash :NAME primary) (gethash :ASSIGNEE primary)))))
              ;; Fork targets = remaining to_stages (all except first, which is primary)
              (let ((fork-targets '()))
                (dolist (ts (rest to-list))
                  (when (hash-table-p ts)
                    (push (cons (gethash :NAME ts) (gethash :ASSIGNEE ts)) fork-targets)))
                (when fork-targets
                  (setf (gethash (cons pipeline-name from) *fork-cache*)
                        (nreverse fork-targets)))))))
        ;; Now build sequential advancement
        ;; Sort stages by order for sequential linking
        (let ((sorted (sort (copy-list stage-list) #'<
                            :key (lambda (s) (or (gethash :ORDER s) 999)))))
          ;; Group by order to handle parallel stages (same order = fork branches)
          ;; For sequential pairs where order(N) < order(N+1), link last of order N to first of order N+1
          ;; But for forks, the from_stage explicitly defines the next
          (loop for (current next) on sorted
                while next
                when (and (hash-table-p current) (hash-table-p next))
                do (let ((cur-name (gethash :NAME current))
                         (cur-order (or (gethash :ORDER current) 0))
                         (next-name (gethash :NAME next))
                         (next-assignee (gethash :ASSIGNEE next))
                         (next-order (or (gethash :ORDER next) 0)))
                     ;; Only link if orders are strictly increasing (skip parallel stages)
                     (when (< cur-order next-order)
                       ;; Check if this stage has a fork override
                       (let ((fork-override (gethash cur-name fork-primary-map)))
                         (if fork-override
                             ;; Fork: use the primary fork target
                             (progn
                               (setf (gethash (cons pipeline-name cur-name) *advancement-cache*)
                                     fork-override)
                               (setf (gethash (cons pipeline-name (car fork-override)) *prev-stage-cache*)
                                     cur-name)
                               ;; Also set prev for all fork targets
                               (let ((fork-tgts (gethash (cons pipeline-name cur-name) *fork-cache*)))
                                 (dolist (ft fork-tgts)
                                   (setf (gethash (cons pipeline-name (car ft)) *prev-stage-cache*)
                                         cur-name))))
                             ;; Normal sequential: cur -> next
                             (progn
                               (setf (gethash (cons pipeline-name cur-name) *advancement-cache*)
                                     (cons next-name next-assignee))
                               (setf (gethash (cons pipeline-name next-name) *prev-stage-cache*)
                                     cur-name)))))
                     ;; Track stage -> pipeline mapping
                     (setf (gethash cur-name *stage-to-pipeline-cache*) pipeline-name)
                     ;; Track participants
                     (setf (gethash (gethash :ASSIGNEE current) participants) t)))
          ;; Handle last stage -> done
          (let ((last (car (last sorted))))
            (when (hash-table-p last)
              (setf (gethash (cons pipeline-name (gethash :NAME last)) *advancement-cache*)
                    (cons "done" (or terminal (gethash :ASSIGNEE last))))
              (setf (gethash (gethash :NAME last) *stage-to-pipeline-cache*) pipeline-name)
              (setf (gethash (gethash :ASSIGNEE last) participants) t)))
          ;; Store participant list
          (setf (gethash pipeline-name *participants-cache*)
                (loop for k being the hash-keys of participants collect k)))))))

(defun reload-pipeline-definitions ()
  "Load all active pipeline definitions from area_content and rebuild caches.
   Called once per tick in the perceive phase. Per D-06: per-tick reload is acceptable
   given 30s-10min tick interval and ~1ms query cost."
  (handler-case
      (progn
        ;; Clear caches
        (clrhash *advancement-cache*)
        (clrhash *prev-stage-cache*)
        (clrhash *stage-to-pipeline-cache*)
        (clrhash *fork-cache*)
        (clrhash *participants-cache*)
        ;; Load and build
        (let ((pipelines (load-pipeline-definitions)))
          (setf *pipeline-cache* pipelines)
          (dolist (meta pipelines)
            (build-advancement-from-pipeline meta))
          (format t "[pipeline] Loaded ~a pipeline definitions from DB~%" (length pipelines))
          pipelines))
    (error (e)
      (format t "[pipeline] Error reloading definitions: ~a~%" e)
      nil)))

;;; ---------------------------------------------------------------------------
;;; Accessor functions — same shape as old defparameters per D-14
;;; ---------------------------------------------------------------------------

(defun get-pipeline-advancement (pipeline-name stage)
  "Return (next-stage . next-assignee) for the given pipeline and stage.
   Per D-07: uses (pipeline, stage) compound key, eliminating stage name collisions.
   Returns nil if stage not found in pipeline."
  (gethash (cons pipeline-name stage) *advancement-cache*))

(defun get-prev-stage (pipeline-name stage)
  "Return the predecessor stage name for the given pipeline and stage.
   Replaces the hardcoded prev-stage-map in action-planner.lisp.
   Returns nil if stage is the first in the pipeline."
  (gethash (cons pipeline-name stage) *prev-stage-cache*))

(defun get-pipeline-type-for-stage (stage)
  "Return pipeline-name for a given stage. Fallback lookup by stage name.
   For stages that appear in multiple pipelines, returns the last-loaded match.
   Prefer get-pipeline-type-for-task (via goal_id chain) for unambiguous lookup."
  (gethash stage *stage-to-pipeline-cache*))

(defun get-fork-targets (pipeline-name stage)
  "Return list of (fork-stage . fork-assignee) for fork points.
   Returns nil if stage is not a fork point."
  (gethash (cons pipeline-name stage) *fork-cache*))

(defun get-pipeline-participants (pipeline-name)
  "Return list of unique assignee strings for a pipeline.
   Used for energy rewards when pipeline completes — scoped to the completing pipeline only
   (fixes existing bug where all pipeline participants got rewarded regardless of involvement)."
  (gethash pipeline-name *participants-cache*))
```
