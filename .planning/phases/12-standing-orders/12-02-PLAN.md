    Inside `phase-rank`, after computing `quality-issue-boost` and before the `urgency` sum (line ~194), add a `schedule-boost` calculation:

    ```lisp
    ;; Standing order schedule boost: +50 when an owned project's cron fires (per D-06, D-07)
    (schedule-boost
      (if (and projects (> (length projects) 0))
          (let ((fired-labels '())
                (now (multiple-value-list (get-decoded-time)))
                ;; get-decoded-time returns: second minute hour date month year dow dst tz
                (current-minute (second now))   ;; index 1
                (current-hour (third now))      ;; index 2 -- BUT these are LOCAL time
                )
            ;; IMPORTANT: get-decoded-time returns LOCAL time, but crons are UTC.
            ;; Use (decode-universal-time (get-universal-time) 0) for UTC.
            (multiple-value-bind (sec min hr date mon yr dow dst tz)
                (decode-universal-time (get-universal-time) 0)
              (declare (ignore sec dst tz))
              (loop for p across (if (vectorp projects) projects (coerce projects 'vector))
                    do (let ((schedule (gethash :schedule p))
                             (pid (gethash :id p)))
                         (when (and schedule
                                    (or (vectorp schedule) (listp schedule))
                                    (> (length schedule) 0))
                           (loop for entry across (if (vectorp schedule) schedule (coerce schedule 'vector))
                                 do (when (hash-table-p entry)
                                      (let ((expr (gethash :expr entry))
                                            (label (gethash :label entry)))
                                        (when (and expr (stringp expr) label)
                                          ;; Check cron match
                                          (when (cron-matches-p expr min hr date mon dow)
                                            ;; Prevent double-fire: check last fire time
                                            (let* ((fire-key (format nil "~a:~a" pid label))
                                                   (last-fire (gethash fire-key *last-schedule-fire*))
                                                   (now-ut (get-universal-time)))
                                              (when (or (null last-fire)
                                                        (> (- now-ut last-fire) 60))
                                                (push label fired-labels)
                                                (setf (gethash fire-key *last-schedule-fire*) now-ut)))))))))))
            ;; Store fired labels for prompt enrichment
            (when fired-labels
              (setf (gethash aid *schedule-fired-labels*) fired-labels)
              (format t "  [schedule] ~a: standing orders fired: ~{~a~^, ~}~%" aid fired-labels))
            (if fired-labels 50 0))
          0))
    ```
