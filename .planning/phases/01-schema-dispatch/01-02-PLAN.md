1. Add a `parse_must_haves(text)` function that extracts must_have truths from PLAN.md. Since `parse_frontmatter()` only does simple key:value, parse must_haves from the markdown body using regex. Look for the `must_haves:` YAML block in frontmatter and extract `truths:` list items:

```python
def parse_must_haves(text):
    """Extract must_have truth strings from PLAN.md frontmatter YAML.
    Handles the nested YAML list format:
      must_haves:
        truths:
          - "Truth one"
          - "Truth two"
    """
    if not text.startswith("---"):
        return []
    end = text.find("\n---", 3)
    if end == -1:
        return []
    fm_block = text[3:end]

    truths = []
    in_truths = False
    for line in fm_block.split('\n'):
        stripped = line.strip()
        if stripped == 'truths:':
            in_truths = True
            continue
        if in_truths:
            if stripped.startswith('- '):
                # Strip quotes and leading "- "
                truth = stripped[2:].strip().strip('"').strip("'")
                truths.append(truth)
            elif stripped and not stripped.startswith('-'):
                # New YAML key — stop parsing truths
                in_truths = False
    return truths
```

```python
def show_status(planning_dir):
    """Show current project and task status from the DB.
    Enhanced per SCHM-05 to show hierarchy and department."""
    conn = get_db()
    cur = conn.cursor(cursor_factory=psycopg2.extras.RealDictCursor)

    cur.execute(
        """SELECT p.id, p.name, p.status, p.owner,
                  COUNT(t.id) FILTER (WHERE t.status IN ('open', 'pending', 'in-progress') AND t.parent_id IS NULL) as open_plans,
                  COUNT(t.id) FILTER (WHERE t.status IN ('done', 'completed') AND t.parent_id IS NULL) as done_plans,
                  COUNT(t.id) FILTER (WHERE t.status IN ('open', 'pending', 'in-progress') AND t.parent_id IS NOT NULL) as open_subtasks,
                  COUNT(t.id) FILTER (WHERE t.status IN ('done', 'completed') AND t.parent_id IS NOT NULL) as done_subtasks
           FROM projects p
           LEFT JOIN tasks t ON t.project_id = p.id AND t.source = 'gsd'
           WHERE p.status = 'active'
           GROUP BY p.id
           ORDER BY p.updated_at DESC""",
    )
    projects = cur.fetchall()

    if not projects:
        print("No active projects in master_chronicle")
    else:
        for p in projects:
            owner = p["owner"] or "unassigned"
            dept = get_owner_department(cur, p["owner"]) if p["owner"] else "—"
            total_plans = p["open_plans"] + p["done_plans"]
            total_subtasks = p["open_subtasks"] + p["done_subtasks"]
            print(f"  [{p['status']}] {p['name']} (owner: {owner}, dept: {dept})")
            print(f"    Plans: {p['done_plans']}/{total_plans} done | Subtasks (must_haves): {p['done_subtasks']}/{total_subtasks} done")

    cur.close()
    conn.close()
```
