```python
def extract_h1_name(text):
    """Extract project name from first H1 heading (# Title) in markdown.
    Per D-04: GSD writes projects with H1 headings, not YAML name fields."""
    for line in text.split('\n'):
        stripped = line.strip()
        if stripped.startswith('# ') and not stripped.startswith('## '):
            return stripped[2:].strip()
    return None
```
