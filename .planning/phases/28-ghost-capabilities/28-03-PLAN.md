    ```lisp
    (defun extract-responsibility-mutations (content)
      "Extract responsibility_mutations JSON array from LLM output content.
       Looks for \"responsibility_mutations\" key in JSON. Returns list of hash-tables or NIL."
      (handler-case
          (let ((key-pos (search "\"responsibility_mutations\"" content)))
            (when key-pos
              (let ((arr-start (position #\[ content :start key-pos)))
                (when arr-start
                  (let ((depth 1) (pos (1+ arr-start)))
                    (loop while (and (< pos (length content)) (> depth 0))
                          do (let ((ch (char content pos)))
                               (cond ((char= ch #\[) (incf depth))
                                     ((char= ch #\]) (decf depth)))
                               (incf pos)))
                    (when (= depth 0)
                      (let* ((json-str (subseq content arr-start pos))
                             (parsed (parse-json json-str)))
                        (when (vectorp parsed)
                          (coerce parsed 'list)))))))))
        (error () nil)))
    ```
