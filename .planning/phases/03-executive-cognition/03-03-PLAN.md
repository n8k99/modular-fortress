   ```lisp
   (defun parse-create-task-lines (content)
     "Extract CREATE_TASK: <description> assignee=<agent_id> lines.
      Returns list of (description assignee-or-nil)."
     (let ((results '())
           (lines (uiop:split-string content :separator '(#\Newline))))
       (dolist (line lines)
         (let ((trimmed (string-trim '(#\Space #\Tab #\Return) line)))
           (when (and (> (length trimmed) 12) (search "CREATE_TASK:" trimmed))
             (let* ((after (string-trim '(#\Space #\Tab)
                            (subseq trimmed (+ (search "CREATE_TASK:" trimmed) 12))))
                    (assignee-pos (search "assignee=" after))
                    (description (string-trim '(#\Space #\Tab)
                                  (if assignee-pos
                                      (subseq after 0 assignee-pos)
                                      after)))
                    (assignee (when assignee-pos
                                (let* ((start (+ assignee-pos 9))
                                       (end (or (position #\Space after :start start)
                                                (length after))))
                                  (subseq after start end)))))
               (when (and description (> (length description) 0))
                 (push (list description assignee) results))))))
       (nreverse results)))
   ```

   - Format: `CREATE_TASK: <description> assignee=<agent_id>` (single line creates AND assigns)

   ```lisp
     "Parse CLASSIFY, DELEGATE, COMPLETE, and CREATE_TASK lines from content and apply via API."
     (let ((classified (parse-classify-lines content))

           (completed (parse-complete-lines content))
   ```
