    **1. Add `persist-pipeline-deliverable` function (new, insert before advance-pipeline):**

    ```lisp
    (defun persist-pipeline-deliverable (pipeline-type task artifact agent-id)
      "Persist the final pipeline deliverable to the appropriate DB table per D-07.
       pipeline-type: one of engineering, editorial, investment, modular-fortress
       artifact: the structured stage artifact (hash-table with schema_version 1)"
      (handler-case
          (let* ((summary (or (gethash :SUMMARY artifact) "Pipeline deliverable"))
                 (title (gethash :TEXT task))
                 (goal-id (gethash :GOAL-ID task))
                 (content (encode-json artifact)))
            (cond
              ;; Engineering + Modular Fortress -> vault_notes
              ((member pipeline-type '("engineering" "modular-fortress") :test #'string-equal)
               (api-post "/api/documents"
                         (json-object :title (format nil "Pipeline Deliverable: ~a" (or title "untitled"))
                                      :content content
                                      :path (format nil "Pipeline/~a/~a" pipeline-type (or goal-id "unknown"))
                                      :tags (json-array "pipeline-deliverable" pipeline-type))))
              ;; Editorial + Investment -> documents
              ((member pipeline-type '("editorial" "investment") :test #'string-equal)
               (api-post "/api/documents"
                         (json-object :title (format nil "~a: ~a" (string-capitalize pipeline-type) (or title "untitled"))
                                      :content content
                                      :path (format nil "Pipeline/~a/~a" pipeline-type (or goal-id "unknown"))
                                      :tags (json-array "pipeline-deliverable" pipeline-type))))
              (t (format t "[persist-deliverable] Unknown pipeline type: ~a, skipping persistence~%" pipeline-type))))
        (error (e)
          (format t "[persist-deliverable] Error persisting to DB: ~a~%" e))))
    ```
