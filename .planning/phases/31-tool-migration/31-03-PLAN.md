```lisp
(defmethod resolve-search ((r noosphere-resolver) search-type terms)
  "Resolve ![type]{key=value} search directives.
   If search-type matches a known tool, invoke it. Otherwise query table."
  ;; Check if this is a tool invocation per D-04
  (let ((tool-def (lookup-tool-definition (string-downcase (symbol-name search-type)))))
    (if tool-def
        ;; Tool dispatch: build args hash from terms, invoke execute-tool-call
        (handler-case
            (let* ((tool-name (string-downcase (symbol-name search-type)))
                   (args (make-hash-table :test #'equal)))
              ;; Convert terms alist to hash-table for execute-tool-call
              (when terms
                (loop for (key . val) in terms
                      do (setf (gethash key args) val)))
              (let ((result (execute-tool-call tool-name args "resolver")))
                (make-innate-result :value result :context :tool)))
          (error (e)
            (make-resistance
             :message (format nil "Tool execution error: ~a" e)
             :source (format nil "~a" search-type))))
        ;; Existing behavior: table lookup
        (let ((table (search-type-to-table search-type)))
          ... existing code unchanged ...))))
```
