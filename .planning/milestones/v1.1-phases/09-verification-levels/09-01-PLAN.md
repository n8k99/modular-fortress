Add a helper function `extract-artifact-issues` before the completion execution block (around line 690). This function takes `stage-notes` (the raw value from task-data :STAGE-NOTES) and returns two values: a list of issue hash-tables and the highest severity string.

```lisp
(defun extract-artifact-issues (stage-notes-raw)
  "Extract issues from structured artifact. Returns (values issues-list highest-severity).
   issues-list is a list of hash-tables with :SEVERITY and :DESCRIPTION.
   highest-severity is CRITICAL > WARNING > SUGGESTION > none."
  (let* ((artifact (cond ((hash-table-p stage-notes-raw) stage-notes-raw)
                          ((stringp stage-notes-raw)
                           (handler-case (parse-json stage-notes-raw)
                             (error () nil)))
                          (t nil)))
         (issues-raw (when (hash-table-p artifact)
                       (gethash :ISSUES artifact)))
         (issues (when issues-raw
                   (cond ((vectorp issues-raw) (coerce issues-raw 'list))
                         ((listp issues-raw) issues-raw)
                         (t nil)))))
    (if (and issues (> (length issues) 0))
        (let* ((has-critical (some (lambda (i) (and (hash-table-p i) (string-equal (gethash :SEVERITY i) "CRITICAL"))) issues))
               (has-warning (some (lambda (i) (and (hash-table-p i) (string-equal (gethash :SEVERITY i) "WARNING"))) issues))
               (has-suggestion (some (lambda (i) (and (hash-table-p i) (string-equal (gethash :SEVERITY i) "SUGGESTION"))) issues))
               (severity-level (cond (has-critical "CRITICAL")
                                     (has-warning "WARNING")
                                     (has-suggestion "SUGGESTION")
                                     (t "none"))))
          (values issues severity-level))
        (values nil "none"))))
```

```lisp
(defun format-quality-assessment (issues severity-level)
  "Format quality assessment section for completion report.
   Per D-08: severity counts + individual CRITICAL/WARNING items listed, SUGGESTION count only."
  (if (or (null issues) (string-equal severity-level "none"))
      "No quality issues found."
      (let* ((critical-items (remove-if-not (lambda (i) (and (hash-table-p i) (string-equal (gethash :SEVERITY i) "CRITICAL"))) issues))
             (warning-items (remove-if-not (lambda (i) (and (hash-table-p i) (string-equal (gethash :SEVERITY i) "WARNING"))) issues))
             (suggestion-count (count-if (lambda (i) (and (hash-table-p i) (string-equal (gethash :SEVERITY i) "SUGGESTION"))) issues))
             (parts '()))
        ;; Build parts list in reverse order (we'll reverse at end)
        (when (> suggestion-count 0)
          (push (format nil "~a suggestion(s)" suggestion-count) parts))
        (dolist (w (reverse warning-items))
          (push (format nil "- WARNING: ~a" (gethash :DESCRIPTION w "")) parts))
        (when (> (length warning-items) 0)
          (push (format nil "~a warning(s):" (length warning-items)) parts))
        (dolist (c (reverse critical-items))
          (push (format nil "- CRITICAL: ~a" (gethash :DESCRIPTION c "")) parts))
        (when (> (length critical-items) 0)
          (push (format nil "~a critical issue(s):" (length critical-items)) parts))
        (push (format nil "**Quality Assessment** [~a]:" severity-level) parts)
        (format nil "~{~a~%~}" parts))))
```
